165  std::pair<bool, T> parse_env(std::string name, T fallback) {
166  auto env_get = std::getenv(name.c_str());
167  try {
168  if(env_get != nullptr) {
169  std::string env_string(env_get);
170  std::stringstream env_stream(env_string);
171  T env_value;
172  env_stream >> env_value;
173  return std::make_pair(true, env_value);
174  } else {
175  return std::make_pair(false, fallback);
176  }
177  } catch (const std::invalid_argument& e) {
178  // environment variable exists, but value is not convertable to an unsigned long
179  std::cerr << msg_illegal_format << name << std::endl;
