257  void first_touch_distribution<instance>::first_touch (const std::size_t& addr) {
258  /* variables for CAS */
259  std::size_t offset, homenode, result;
260  static const std::size_t rank = argo::backend::node_id();
261  static const std::size_t global_null = base_distribution<instance>::total_size + 1;
262  const std::size_t owners_dir_window_index = page_info_size * (addr / granularity);
263  /* decentralize CAS */
264  const std::size_t cas_node = (addr / granularity) % base_distribution<instance>::nodes;
265 
266  /* check/try to acquire ownership of the page with CAS to process' cas_node index */
267  argo::backend::atomic::_compare_exchange_owners_dir(&rank, &global_null, &result, sizeof(std::size_t), cas_node, owners_dir_window_index+2);
268 
269  /* this process was the first one to deposit its rank */
270  if (result == global_null) {
271  /* iterate through the nodes to find a valid offset for the page, starting from the currently running one */
272  bool succeeded = false;
273  std::size_t n, searched;
274  for(n = rank, searched = 0; searched < static_cast<std::size_t>(base_distribution<instance>::nodes);
275  n = (n + 1) % base_distribution<instance>::nodes, searched++) {
276  /* load backing offset for the node from the offsets table */
277  argo::backend::atomic::_load_local_offsets_tbl(&offset, rank, n);
278  /* check if the backing store size of this node is sufficient */
279  while (offset < base_distribution<instance>::size_per_node) {
280  /* try to claim a valid offset */
281  const std::size_t incr_offset = offset + granularity;
282  argo::backend::atomic::_compare_exchange_offsets_tbl(&incr_offset, &offset, &result, sizeof(std::size_t), n, n);
283  if (result == offset) { succeeded = true; homenode = n; break; }
284  offset = result;
285  }
286  /* cache the offset returned from a remote CAS operation */
287  if (n != rank) {
288  argo::backend::atomic::_store_local_offsets_tbl(offset, rank, n);
289  }
290  /* succeeded, leave */
291  if (succeeded) break;
292  }
293 
294  /* this should never happen */
295  if (!succeeded) {
296  std::cerr << msg_first_touch_fail << std::endl;
297  throw std::system_error(std::make_error_code(static_cast<std::errc>(errno)), msg_first_touch_fail);
298  exit(EXIT_FAILURE);
299  }
300 
301  /* store page info in the local window */
302  std::size_t page_info[page_info_size] = {homenode, offset, rank};
303  argo::backend::atomic::_store_local_owners_dir(page_info, page_info_size, rank, owners_dir_window_index);
304 
305  /* store page info in the remote public cas_node window */
306  if (rank != cas_node) {
307  argo::backend::atomic::_store_public_owners_dir(page_info, sizeof(std::size_t), page_info_size, cas_node, owners_dir_window_index);
308  }
309  } else {
310  /* load page info from the remote public cas_node window */
311  if (rank != cas_node) {
312  std::size_t page_info[page_info_size];
313  do {
314  argo::backend::atomic::_load_public_owners_dir(page_info, sizeof(std::size_t), page_info_size, cas_node, owners_dir_window_index);
315  } while (is_one_equal_to(page_info, global_null));
316  /* store page info in the local window */
317  argo::backend::atomic::_store_local_owners_dir(page_info, page_info_size, rank, owners_dir_window_index);
318  }
319  }
