501  sem_post(&ibsem);
502  return;

507  i < start_index+load_size;

663  }
664  }

668  MPI_Win_lock(MPI_LOCK_EXCLUSIVE, owner, 0, sharerWindow);
669  MPI_Accumulate(sharer_bit_mask.data(), classification_size, MPI_LONG, owner,
670  classification_index_array[0], classification_size, MPI_LONG,
671  MPI_BOR, sharerWindow);
672  MPI_Win_unlock(owner, sharerWindow);
673  }
674  }
675  }

678  MPI_Win_lock(MPI_LOCK_SHARED, load_node , 0, globalDataWindow[load_node]);
679  MPI_Get(temp_data.data(), fetch_size, cacheblock,
680  load_node, load_offset, fetch_size, cacheblock, globalDataWindow[load_node]);
681  MPI_Win_unlock(load_node, globalDataWindow[load_node]);

685  /* Update only the pages necessary */

691  void* temp_ptr = static_cast<char*>(startAddr) + temp_addr;

699  mprotect(temp_ptr, block_size, PROT_READ);
700  }
701  touchedcache[idx] = 1;
702  cacheControl[idx].state = VALID;
703  cacheControl[idx].dirty=CLEAN;
