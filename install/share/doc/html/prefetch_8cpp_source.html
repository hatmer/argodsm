106  char *tmp = static_cast<char*>(collective_alloc(alloc_size));
107  std::size_t num_nodes = argo::number_of_nodes();

110 
111  std::size_t stride, start_page;
112  /* Figure out boundaries for both block_size and load_size */
113  if(dd::is_cyclic_policy()) {
114  /*
115  * For blocked policies, at most one block is guaranteed
116  * to be contiguious in both global memory and backing store
117  */
118  stride = std::min(block_size, load_size);
119  start_page = block_size-1; //first block may already be fetched in init
120  } else {
121  /*
122  * For the first touch policiy, at most
123  * (alloc_size/4096)/num_nodes are guaranteed to be contiguious
124  * in both global memory and backing store.
125  */
126  stride = (load_size < ((size/page_size)/num_nodes)) ?
127  load_size : (size/page_size)/num_nodes - 1;
128  start_page = stride-1; //First stride may already be fetched in init
129  }
130  std::size_t end_page = start_page+stride;
131
