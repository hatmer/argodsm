50  auto old = backend::atomic::exchange(lastuser, locked, atomic::memory_order::relaxed);
51  if(old != locked) {
52  std::size_t self = backend::node_id();
53  if(old == self || old == init) {
54  /* note: doing nothing here is only safe because we are using
55  * an SC for DRF memory model in ArgoDSM.
56  * When changing this to something more strict, e.g.
57  * TSO, then here a write buffer ordering must be
58  * enforced.
59  * A trivial implementation would call a
60  * self-downgrade (as release() does), but a
61  * better implementation could be thought of
62  * if a better write-buffer is also implemented.
63  * why: semantically, we acquire at the beginning of
64  * a lock. Any OTHER node seeing changes from within
65  * the critical section can could therefore deduce
66  * which writes from before the critical section must
67  * have been issued. As write buffers can be cleared at
68  * any time without ordering guarantees, this may cause
69  * problems depending on the memory model. Using
70  * SC for DRF disallows making such deductions, as they
71  * would imply a data race.
72  */
73

75  * The global lock still has to function as a correct
76  * lock locally, so semantically we must ensure proper
77  * synchronization at least within this node.
78  */
