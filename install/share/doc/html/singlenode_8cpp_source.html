131  void barrier(std::size_t threadcount) {
132  /* initially: flag = false */
133  std::unique_lock<std::mutex> barrier_lock(barrier_mutex);
134  barrier_counter++;
135  if(barrier_counter == threadcount) {
136  /* avoid progress if some thread is still stuck in the previous barrier */
137  barrier_cv.wait(barrier_lock, []{ return !barrier_flag.load(); });
138  /* all threads in this barrier now,
139  * flip flag to avoid progress in next barrier */
140  barrier_flag = true;
141  barrier_cv.notify_all();
142  }
143 
144  /* wait for flag to flip to ensure all threads have reached this barrier */
145  barrier_cv.wait(barrier_lock, []{ return barrier_flag.load(); });
146 
147  /* all threads synchronized here. cleanup needed. */
148 
149  barrier_counter--;
150  if(barrier_counter == 0) {
151  /* all threads successfully left the wait call, allow next barrier */
152  barrier_flag = false;
153  barrier_cv.notify_all(); // other threads + unstick next barrier's final thread
154  }
155 
160  barrier_cv.wait(barrier_lock, []{ return !barrier_flag; });
161  }
